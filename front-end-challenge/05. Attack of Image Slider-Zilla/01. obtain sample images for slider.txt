jQuery(document).ready(function($){
  var $icons, svg_icons
  $icons = $('li.icon[data-id] > img[width="64"]')
  svg_icons = []
  $icons.each(function(){
    var $icon, title, png_url, svg_url
    $icon = $(this)
    title = $icon.attr('title')
    png_url = $icon.attr('src')
    svg_url = png_url.replace(/icons\/png\/\d+/, 'icons/svg').replace(/\.png$/, '.svg')
    svg_icons.push([title, svg_url])
  })
  console.log( JSON.stringify(svg_icons))
})
